```python
async def detect_language(self, text: str) -> LanguageDetectionResult:
    """Detect language from transcribed text using langdetect library."""
    try:
        detected_language = langdetect_detect(text)

        # Map to supported language if possible; default to detected
        if detected_language not in self.supported_languages:
            self.logger.info(
                f"Detected '{detected_language}' not in supported list, returning as-is"
            )

        confidence = 0.95  # langdetect does not provide confidence for detect()

        # Detect accent if supported
        accent_region = None
        dialect = None
        if detected_language in self.accent_regions:
            # langdetect cannot determine accent; leave as None
            accent_region = None
            dialect = None

        return LanguageDetectionResult(
            language=detected_language,
            confidence=confidence,
            dialect=dialect,
            accent_region=accent_region,
        )

    except Exception as e:
        self.logger.warning(f"Language detection failed ({e}), defaulting to 'en'")
        return LanguageDetectionResult(
            language="en",
            confidence=0.5,
            dialect=None,
            accent_region=None,
        )
```
