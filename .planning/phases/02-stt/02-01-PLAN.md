    ```python
    import os
    from typing import Optional
    from deepgram import Deepgram
    from ..agents.stt_agent import TranscriptionResult

    class DeepgramProvider:
        """Deepgram SDK integration for speech-to-text"""
        
        def __init__(self, api_key: Optional[str] = None):
            self.api_key = api_key or os.getenv("DEEPGRAM_API_KEY")
            self._client: Optional[Deepgram] = None
            
        async def initialize(self) -> None:
            """Initialize Deepgram client"""
            if not self.api_key:
                raise ConfigurationError("DEEPGRAM_API_KEY is required")
            self._client = Deepgram(self.api_key)
            
        async def transcribe(self, audio_data: bytes, sample_rate: int = 16000) -> TranscriptionResult:
            """Transcribe audio using Deepgram"""
            if not self._client:
                await self.initialize()
                
            response = await self._client.transcription.prerecorded(
                {"buffer": audio_data, "mimetype": "audio/wav"},
                {"model": "nova-2", "smart_format": True}
            )
            
            # Parse response and create TranscriptionResult
            result = response["results"]["channels"][0]["alternatives"][0]
            return TranscriptionResult(
                text=result["transcript"],
                confidence=result.get("confidence", 0.0),
                language=result.get("language", "en"),
                dialect=None,
                timestamps=[],
                is_partial=False,
                chunk_id=0
            )
            
        async def close(self) -> None:
            """Clean up resources"""
            self._client = None
    ```

    ```python
    async def transcribe(self, chunk: AudioChunk) -> TranscriptionResult:
        """Transcribe audio chunk to text"""
        
        # Check for Deepgram first
        if os.getenv("DEEPGRAM_API_KEY"):
            try:
                provider = DeepgramProvider()
                await provider.initialize()
                audio_bytes = self._audio_to_bytes(chunk)
                result = await provider.transcribe(audio_bytes, chunk.sample_rate)
                return result
            except Exception as e:
                self.logger.warning(f"Deepgram failed: {e}, falling back to Voxtral")
                # Fall through to Voxtral
        
        # Use Voxtral as primary or fallback
        return await self._transcribe_with_voxtral(chunk)
    ```
