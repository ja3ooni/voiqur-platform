2. **Expand `generate_response()`** to accept an optional `tool_executor` parameter and handle the tool call loop. The updated signature and body:
   ```python
   async def generate_response(
       self,
       context: ConversationContext,
       system_prompt: Optional[str] = None,
       tools: Optional[List[Dict]] = None,
       tool_executor: Optional[Any] = None,
   ) -> str:
       """Generate response using Mistral API with optional tool calling."""
       try:
           messages = []
           if system_prompt:
               messages.append({"role": "system", "content": system_prompt})
           messages.extend(context.get_context_for_model())

           kwargs: Dict[str, Any] = {
               "model": self._model,
               "messages": messages,
           }
           if tools:
               kwargs["tools"] = tools
               kwargs["tool_choice"] = "auto"

           response = await self._client.chat.complete_async(**kwargs)
           msg = response.choices[0].message

           # Tool call loop: if Mistral returns tool_calls, execute and re-call
           if msg.tool_calls and tool_executor is not None:
               # Append assistant message WITH tool_calls to history (Pitfall 2)
               assistant_dict = msg.model_dump(exclude_none=True) if hasattr(msg, "model_dump") else {
                   "role": "assistant",
                   "content": msg.content or "",
                   "tool_calls": [tc.model_dump(exclude_none=True) if hasattr(tc, "model_dump") else {"id": tc.id, "function": {"name": tc.function.name, "arguments": tc.function.arguments}} for tc in msg.tool_calls],
               }
               if "role" not in assistant_dict:
                   assistant_dict["role"] = "assistant"
               messages.append(assistant_dict)

               for tc in msg.tool_calls:
                   fn_name = tc.function.name
                   fn_args = json.loads(tc.function.arguments) if isinstance(tc.function.arguments, str) else tc.function.arguments

                   # Execute via existing ToolExecutor
                   from .tool_integration import ToolCall as TIToolCall, ToolStatus
                   ti_call = TIToolCall(tool_name=fn_name, parameters=fn_args, caller_agent_id="llm_agent")
                   executed = await tool_executor.execute_tool_call(ti_call)

                   result_str = str(executed.result) if executed.status == ToolStatus.COMPLETED else f"Error: {executed.error_message}"

                   messages.append({
                       "role": "tool",
                       "name": fn_name,
                       "content": result_str,
                       "tool_call_id": tc.id,
                   })

               # Re-call Mistral with tool results (no tools param — let it generate text)
               final_response = await self._client.chat.complete_async(
                   model=self._model,
                   messages=messages,
               )
               content = final_response.choices[0].message.content
               return content if content else ""

           content = msg.content
           return content if content else ""

       except Exception as e:
           self.logger.error(f"Mistral API call failed: {e}")
           raise
   ```
