```python
"""Integration tests for multi-turn LLM conversation with tool calling (LLM-05)."""
import json
import pytest
from unittest.mock import AsyncMock, MagicMock, patch, PropertyMock
from src.agents.llm_agent import (
    MistralModelManager,
    ConversationContext,
    get_mistral_tools,
)
from src.agents.tool_integration import (
    ToolRegistry,
    ToolExecutor,
    ToolParameter,
    ToolCall as TIToolCall,
    ToolStatus,
)


def _make_chat_response(content, tool_calls=None):
    """Helper: build a mock Mistral chat response."""
    msg = MagicMock()
    msg.content = content
    msg.tool_calls = tool_calls
    if tool_calls is None:
        msg.tool_calls = None
    # model_dump for serialization in the tool loop
    msg.model_dump = MagicMock(return_value={
        "role": "assistant",
        "content": content or "",
        **({"tool_calls": [
            {"id": tc.id, "function": {"name": tc.function.name, "arguments": tc.function.arguments}}
            for tc in (tool_calls or [])
        ]} if tool_calls else {}),
    })
    resp = MagicMock()
    resp.choices = [MagicMock(message=msg)]
    return resp


def _make_tool_call(call_id, fn_name, fn_args_dict):
    """Helper: build a mock Mistral SDK ToolCall object."""
    tc = MagicMock()
    tc.id = call_id
    tc.function = MagicMock()
    tc.function.name = fn_name
    tc.function.arguments = json.dumps(fn_args_dict)
    tc.model_dump = MagicMock(return_value={
        "id": call_id,
        "function": {"name": fn_name, "arguments": json.dumps(fn_args_dict)},
    })
    return tc


@patch("src.agents.llm_agent.Mistral")
async def test_multi_turn_tool_conversation(MockMistral, mock_mistral_env):
    """LLM-05: 3-turn conversation with tool call execution verified end-to-end."""

    # --- Setup mock Mistral client ---
    mock_client = MagicMock()

    # Turn 1: Mistral wants to call get_current_time tool
    tool_call_obj = _make_tool_call("tc-001", "get_current_time", {"timezone": "UTC"})
    turn1_response = _make_chat_response(content=None, tool_calls=[tool_call_obj])

    # Turn 1 follow-up: After receiving tool result, Mistral returns text
    turn1_followup = _make_chat_response(content="The current time in UTC is 2026-03-28 14:00:00.")

    # Turn 2: User asks follow-up, Mistral answers with context
    turn2_response = _make_chat_response(content="That was the time I just told you — 2:00 PM UTC.")

    mock_client.chat.complete_async = AsyncMock(
        side_effect=[turn1_response, turn1_followup, turn2_response]
    )
    MockMistral.return_value = mock_client

    # --- Setup real ToolRegistry + ToolExecutor with a simple time tool ---
    registry = ToolRegistry()

    async def mock_get_time(timezone=None):
        return f"Current time in {timezone or 'local'}: 2026-03-28 14:00:00"

    registry.register_function(
        name="get_current_time",
        description="Get the current date and time",
        function=mock_get_time,
        parameters=[
            ToolParameter(name="timezone", type="string", description="TZ", required=False)
        ],
    )
    executor = ToolExecutor(registry)

    # --- Build conversation ---
    mgr = MistralModelManager()
    ctx = ConversationContext(session_id="integration-test")
    tools = get_mistral_tools(registry)

    # Turn 1: user asks time — triggers tool call loop
    ctx.add_message("user", "What time is it in UTC?")
    response1 = await mgr.generate_response(
        ctx, system_prompt="You are a helpful assistant.", tools=tools, tool_executor=executor
    )
    ctx.add_message("assistant", response1)

    assert response1, "Turn 1 response must be non-empty"
    assert "14:00" in response1 or "time" in response1.lower(), (
        f"Turn 1 should reference the time from tool result, got: {response1}"
    )

    # Turn 2: follow-up question — should use conversation context
    ctx.add_message("user", "What time was that again?")
    response2 = await mgr.generate_response(
        ctx, system_prompt="You are a helpful assistant.", tools=tools, tool_executor=executor
    )
    ctx.add_message("assistant", response2)

    assert response2, "Turn 2 response must be non-empty"

    # --- Verify call counts ---
    # complete_async called: (1) initial with tools, (2) after tool result, (3) turn 2
    assert mock_client.chat.complete_async.call_count == 3, (
        f"Expected 3 Mistral API calls, got {mock_client.chat.complete_async.call_count}"
    )

    # Verify the second call (after tool result) included tool result messages
    second_call_kwargs = mock_client.chat.complete_async.call_args_list[1]
    second_msgs = second_call_kwargs.kwargs.get("messages", [])
    tool_msgs = [m for m in second_msgs if m.get("role") == "tool"]
    assert len(tool_msgs) >= 1, "Tool result message must be in the second Mistral call"

    # Verify the third call has multi-turn history (at least 2 user messages)
    third_call_kwargs = mock_client.chat.complete_async.call_args_list[2]
    third_msgs = third_call_kwargs.kwargs.get("messages", [])
    user_msgs = [m for m in third_msgs if m.get("role") == "user"]
    assert len(user_msgs) >= 2, f"Turn 2 should have full history with >=2 user messages, got {len(user_msgs)}"


@patch("src.agents.llm_agent.Mistral")
async def test_no_tool_calls_returns_text_directly(MockMistral, mock_mistral_env):
    """When Mistral returns no tool_calls, generate_response returns content directly."""
    mock_client = MagicMock()
    mock_client.chat.complete_async = AsyncMock(
        return_value=_make_chat_response(content="Just a plain answer.")
    )
    MockMistral.return_value = mock_client

    mgr = MistralModelManager()
    ctx = ConversationContext(session_id="no-tools")
    ctx.add_message("user", "Hi there")
    result = await mgr.generate_response(ctx)

    assert result == "Just a plain answer."
    assert mock_client.chat.complete_async.call_count == 1
```

Key design choices:
- Uses real `ToolRegistry` + `ToolExecutor` with a mock function (not mocking the executor itself) so the integration path through `execute_tool_call` is exercised.
- Mocks only the Mistral API (via `@patch("src.agents.llm_agent.Mistral")`) to avoid real HTTP calls.
- Verifies call count, message history content, and tool result injection.
- Second test (`test_no_tool_calls_returns_text_directly`) confirms the non-tool path still works.
