   async def synthesize(self, request: SynthesisRequest) -> SynthesisResult:
       if not self.model_loaded:
           raise RuntimeError("XTTS-v2 model not loaded")
       start_time = time.time()
       try:
           voice_model = self.voice_models.get(request.voice_id)
           if not voice_model:
               raise ValueError(f"Voice model {request.voice_id} not found")
           if request.language not in self.supported_languages:

           # Call ElevenLabs SDK v2: text_to_speech.convert()
           audio_bytes_iter = await self.client.text_to_speech.convert(
               text=request.text,

               }
           )
