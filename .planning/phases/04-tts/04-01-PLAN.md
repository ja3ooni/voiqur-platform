```python
async def synthesize(self, request: SynthesisRequest) -> SynthesisResult:
    """Synthesize speech using ElevenLabs cloud API"""
    if not self.model_loaded:
        raise RuntimeError("XTTS-v2 model not loaded")
    
    api_key = os.getenv("ELEVENLABS_API_KEY")
    if not api_key:
        raise RuntimeError("ELEVENLABS_API_KEY not configured")
    
    start_time = time.time()
    
    # Get voice - use default or request-specified
    voice_id = request.voice_id or os.getenv("ELEVENLABS_VOICE_ID", "Rachel")
    
    # Generate with ElevenLabs
    try:
        audio = generate(
            text=request.text,
            voice=voice_id,
            model="eleven_multilingual_v2",
            api_key=api_key
        )
        
        # Convert bytes to numpy array
        audio_array = np.frombuffer(audio, dtype=np.int16)
        audio_float = audio_array.astype(np.float32) / 32768.0
        
        processing_time = time.time() - start_time
        
        # Get sample rate (ElevenLabs uses 44100)
        sample_rate = 44100
        
        return SynthesisResult(
            audio_data=audio_float,
            sample_rate=sample_rate,
            duration=len(audio_float) / sample_rate,
            voice_id=voice_id,
            text=request.text,
            quality_score=4.5,  # ElevenLabs default MOS
            processing_time=processing_time,
            metadata={
                "emotion": request.emotion.value,
                "language": request.language,
                "provider": "elevenlabs"
            }
        )
    except Exception as e:
        self.logger.error(f"ElevenLabs synthesis failed: {e}")
        raise
```
