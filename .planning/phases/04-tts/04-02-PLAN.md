```python
def _get_coqui_model(self) -> CoquiTTS:
    """Lazy-load Coqui TTS model"""
    if not hasattr(self, '_coqui_tts'):
        # Load multilingual XTTS-v2 model
        self._coqui_tts = CoquiTTS(
            model_name="tts_models/multilingual/mxtts/multilingual_fastscan",
            gpu=True if torch.cuda.is_available() else False
        )
    return self._coqui_tts
```

3. Add fallback logic to synthesize:
```python
async def synthesize(self, request: SynthesisRequest) -> SynthesisResult:
    # ... ElevenLabs code first ...
    
    # FALLBACK: If ElevenLabs fails or not configured
    if not api_key or fallback_requested:
        try:
            coqui = self._get_coqui_model()
            
            # Generate with Coqui TTS
            wav = coqui.tts(
                text=request.text,
                speaker_wav=None,  # Use default speaker
                language=self._map_language(request.language)
            )
            
            audio_float = wav.squeeze()
            sample_rate = 24000
            
            return SynthesisResult(...)
        except Exception as e:
            self.logger.error(f"TTS synthesis failed: {e}")
            raise
```

4. Add language mapping:
```python
def _map_language(self, lang: str) -> str:
    """Map ISO code to Coqui format"""
    mapping = {"en": "en", "ar": "ar", "fr": "fr", "de": "de", "es": "es"}
    return mapping.get(lang, "en")
```
