Critical pitfall (from RESEARCH.md): load_dotenv() MUST be called before APIConfig() is instantiated. APIConfig uses default_factory which evaluates os.getenv() at construction time. If load_dotenv() comes after, .env values are ignored.

def main():
    """Main entry point for the API server."""
    from dotenv import load_dotenv
    load_dotenv()  # Must be before APIConfig() — reads .env into os.environ

    app = create_app(config)
